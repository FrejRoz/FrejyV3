//@version=5
indicator("BB Entry with Exact Liq Hunter POC - Confirmed", overlay=true)

// --- Inputs ---
groupTime = "TIME FILTER"
useDateFilter = input.bool(true, "Use Start Date Filter", group=groupTime)
timeZone      = input.string("America/New_York", "Timezone", group=groupTime)
startYear     = input.int(2026, "Start Year", group=groupTime)
startMonth    = input.int(5, "Start Month", group=groupTime)
startDay      = input.int(13, "Start Day", group=groupTime)
startHour     = input.int(3, "Start Hour", group=groupTime)
startMin      = input.int(15, "Start Minute", group=groupTime)

groupStrat = "STRATEGY SETTINGS"
tradeSide  = input.string("Both", "Trade Side", options=["Long", "Short", "Both"], group=groupStrat)
entryMode  = input.string("Bottom to Center", "Entry Mode", options=["Bottom Only", "Bottom to Center"], group=groupStrat)
bbLength   = input.int(20, "BB Length", group=groupStrat)
bbMult     = input.float(2.0, "BB StdDev", group=groupStrat)

groupPOC = "POC FILTER SETTINGS"
i_vpLen    = input.int(50, "POC Lookback Bars", minval=10, group=groupPOC)
i_vpRows   = input.int(24, "POC Price Rows", minval=8, group=groupPOC)

// --- Time Logic ---
startTime = timestamp(timeZone, startYear, startMonth, startDay, startHour, startMin)
inDate    = useDateFilter ? (time >= startTime) : true

// --- EXACT LIQ HUNTER POC CALCULATION ---
f_calculatePOC() =>
    float pHi = ta.highest(high, i_vpLen)
    float pLo = ta.lowest(low,   i_vpLen)
    float pRng = pHi - pLo
    float resPOC = na
    if pRng > 0
        float rowH = pRng / i_vpRows
        array<float> rowVol = array.new<float>(i_vpRows, 0.0)
        for j = 0 to i_vpLen - 1
            float tp = (high[j] + low[j] + close[j]) / 3.0
            int row = math.floor((tp - pLo) / rowH)
            row := math.max(0, math.min(row, i_vpRows - 1))
            array.set(rowVol, row, array.get(rowVol, row) + volume[j])
        int   pocIdx = 0
        float maxV = array.get(rowVol, 0)
        for k = 1 to i_vpRows - 1
            float rv = array.get(rowVol, k)
            if rv > maxV
                maxV := rv
                pocIdx := k
        resPOC := pLo + (pocIdx + 0.5) * rowH
    resPOC

float dynamicPOC = f_calculatePOC()

// --- POC Bias Logic ---
bool isAbovePOC = not na(dynamicPOC) and close > dynamicPOC
bool isBelowPOC = not na(dynamicPOC) and close < dynamicPOC

// --- Indicators (BB) ---
[middle, upper, lower] = ta.bb(close, bbLength, bbMult)

// --- State Tracking ---
var bool isBuying  = false
var bool isSelling = false
var float totalBuyCost = 0.0
var int buyCount = 0

if inDate
    if low <= lower
        isBuying := true
    if close >= middle
        isBuying := false
        totalBuyCost := 0.0 
        buyCount := 0
    if high >= upper
        isSelling := true
    if close <= middle
        isSelling := false

// --- Signal Logic ---
canLong  = false
canShort = false

if inDate
    if (tradeSide == "Long" or tradeSide == "Both") and isAbovePOC
        canLong := (entryMode == "Bottom Only") ? low <= lower : (isBuying and (low <= lower or close < middle))
    
    if (tradeSide == "Short" or tradeSide == "Both") and isBelowPOC
        canShort := (entryMode == "Bottom Only") ? high >= upper : (isSelling and (high >= upper or close > middle))

// --- Avg Entry Calc ---
if canLong
    totalBuyCost += close
    buyCount += 1
float avgEntry = buyCount > 0 ? totalBuyCost / buyCount : na

// --- Confirmed Signals (Label Control) ---
// Sinyallerin sadece mum kapandığında (confirmed) görünmesini sağlar.
bool confirmedLong  = canLong and barstate.isconfirmed
bool confirmedShort = canShort and barstate.isconfirmed

// --- Plotting ---
plot(dynamicPOC, "Liq Hunter POC", color=color.new(color.yellow, 30), linewidth=2)

// Sinyal Etiketleri (Sadece Mum Kapandığında Basılır)
plotshape(confirmedLong,  title="Confirmed Buy",  style=shape.labelup,   location=location.belowbar, color=color.green, text="$",  textcolor=color.white)
plotshape(confirmedShort, title="Confirmed Sell", style=shape.labeldown, location=location.abovebar, color=color.red,   text="$", textcolor=color.white)

// BB Kanalları
plot(middle, color=color.gray, title="BB Middle")
plot(upper,  color=color.red,  title="BB Upper")
plot(lower,  color=color.green, title="BB Lower")

// --- ALERTS ---
// Alarmlar zaten genellikle mum kapanışında kurulur, 
// ancak burada sinyalleri direkt 'confirmed' değişkenine bağladık.
alertcondition(confirmedLong, title="Buy Alert", message="Buy Signal Confirmed on Close!")
alertcondition(confirmedShort, title="Sell Alert", message="Sell Signal Confirmed on Close!")
